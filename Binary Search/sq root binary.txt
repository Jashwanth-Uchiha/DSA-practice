class Solution {
public:
    int floorSqrt(int n)  {
     int start=0;int end=n;long long  mid =start+(end-start)/2;
      while(start<=end){
         mid =start+(end-start)/2;
         if((mid*mid)>n){
            end=mid ;
         }
         else if((mid*mid)<n){
            if((mid+1)*(mid+1)>n&&mid!=n){
                return mid;
            }
           start=mid+1;
         }
         else{
            return mid;
         }
      }
      return mid;
    }
};
