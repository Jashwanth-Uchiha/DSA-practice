class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
         int start=0; int end =nums.size()-1;
        for(int i=0;i<= end;i++){
            for(int j =i+1;j<=end;j++){
                if(nums[i]+nums[j]==target){
                    return {i,j};
                }
            }
        }
        return {-1,-1};
    }
};
