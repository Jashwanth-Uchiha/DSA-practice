class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
         int maxi = 0;
    for (int i=0; i<nums.size();i++) {
        long long sum=0;
        for (int j=i;j<nums.size();j++) {
            sum=sum+nums[j];
            if (sum==k){maxi++;
            }
        }
    }
    return maxi; 

        }
    
};
