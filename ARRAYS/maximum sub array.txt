class Solution {
public:
    int maxSubArray(vector<int>& nums) {int sum=0;int maxi=INT_MIN;int count=0;

        for(int i=0;i<nums.size();i++){
             sum=sum+nums[i];
             if(sum<0){
                sum=0;count++;
             }
             maxi=max(sum,maxi);
        }
        if(count==nums.size()){
            return *max_element(nums.begin(),nums.end());
        }
        return maxi;
    }
};
