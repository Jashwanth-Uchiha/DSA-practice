class Solution {
public:
    void sortColors(vector<int>& nums) {
       int j=0;int count=0;
       for(int i=0;i<nums.size();i++){
        if(nums[i]!=2){
            swap(nums[j++],nums[i]);
        }
       }
        for(int i=0;i<nums.size();i++){
           if(nums[i]==2){
            count ++;
           } 
        }
        int k=0;
       for(int i=0;i<nums.size()-count;i++){
        if(nums[i]!=1){
            swap(nums[k++],nums[i]);
        }
       }
    }
};
