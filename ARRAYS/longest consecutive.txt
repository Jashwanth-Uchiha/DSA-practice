class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
     sort(nums.begin(),nums.end());int count=1,maxi=INT_MIN;
     for(int i=0;i<nums.size()-1;i++){
          if(nums[i+1]-nums[i]==1){
            count ++; maxi=max(count,maxi);
          }
          else if(nums[i+1]-nums[i]==0){
            continue;
          }
          else{
     count=0;
          }  
     }
     return maxi;
    }
};
