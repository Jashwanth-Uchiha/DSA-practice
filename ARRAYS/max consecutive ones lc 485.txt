class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
   
        int count = 0,max =INT_MIN ;
        for(int x: nums){
            if(x==1){
                count++;
            }
            else{
                count=0;
            }

            if(count>max){
                max=count;
            }
        }

        return max;
    }
};
