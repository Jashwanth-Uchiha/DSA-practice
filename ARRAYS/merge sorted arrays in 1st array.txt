class Solution {
public:
    void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
      for(int j=0;j<n;j++){
        nums1[m+j]=nums2[j];
      }
      
     for(int i=0;i<m+n;i++){
        if( i<m+n-1 &&nums1[i]>nums1[i+1]){
            swap(nums1[i],nums1[i+1]);
            i=-1;
        }
     }
    }
};
